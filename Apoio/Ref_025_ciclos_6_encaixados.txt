#include <stdio.h>

// Programa Ref_025_ciclos_6_encaixados

// Ciclos encaixados

// são ciclos dentro de ciclos
// sempre que um ciclo encaixado é iniciado somente termina
// o seu processo repetitivo quando a sua condição deixar de
// ser satisfeita, voltando o controlo para o ciclo externo
//

int main()
{
	printf("\n\nUsando while\n\n");
	int i, k;
	
	i = 0;
	while (i < 2)
	{
		printf("\n\nViva\n");
		k = 0;
		while(k < 3)
		{
			printf(" Lumiar");
			k++;// atualiza ctdr ciclo interno
		}

		i++; // atualiza ctdr ciclo externo
	}

// usando ciclo for

	printf("\n\n\nUsando for\n\n");

	for (i = 0; i < 2; i++)
	{
		printf("\n\nViva\n");

		for(k = 0; k < 3; k++)
		{
			printf(" Tejo");
		}
	}

	printf("\n\n");
	system("PAUSE");

	return 0; 
}
