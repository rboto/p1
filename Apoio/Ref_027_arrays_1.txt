#include <stdio.h>
#include <stdlib.h>

#define NUM_ELS 5

// Programa Ref_027_arrays_1

// Introdução aos arrays 
// Um array é conjunto de els do mesmo tipo de dados
// Tal como as strings cada elemento é acedido através 
// do nome do array e de um índice de posição, a partir
// da posicao 0

// Todavia, ao contrário das strings os arrays não têm
// um indicador do seu fim

// Este programa calcula a média de cinco temperaturas
// digitadas para um array

int main()
{
	float temps[NUM_ELS]; // declara um array	
	float soma, media;
	int i;

// Carregar na M as temps
	i = 0;
	while(i < NUM_ELS)
	{
		printf("\n\nDigite a temperatura %d: ", i);
		fflush(stdin);	
		scanf("%f", &temps[i]);

		i++;
	}

// Ver os valores da M
	printf("\n\nEis os valores digitados: \n\n");
	i = 0;
	while(i < NUM_ELS)
	{	
		printf("\n%4.2f", temps[i]);

		i++;
	}

// Somar as temps
	soma = 0;
	i = 0;
	while(i < NUM_ELS)
	{	
		soma = soma +  temps[i];
		i++;
	}

// Media

	media = soma / NUM_ELS;
	printf("\n\nA Media = %4.2f", media);
	
	printf("\n\nDigite um carater para continuar");
	fflush(stdin);	
	getchar();

	return 0;
} // fecha o main
