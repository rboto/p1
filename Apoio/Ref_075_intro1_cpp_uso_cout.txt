#include <stdio.h>
#include <iostream>
#include <stdlib.h>

using namespace std;


// Prgrama Ref_075_intro1_cpp_uso_cout

/* 
   Uso da primitiva cout

   Programa Boas Vindas em C++

   C++ é uma Linguagem baseada e orientada para objectos

   As classes permitem o encapsulamento (information hiding)
   da informação e a reutilização de software.

   Um exemplo disto é aplicado neste programa através do 
   método cout que pertence ao espaço de nomes std.

   cout é a primitiva equivalente à função printf (C puro),
   só que cout deteta implicitamente os tipos dos dados a
   imprimir, pelo que dispensa assim a sua especificação.

   cout significa "insere no stream".

*/


/*
  O comando abaixo é uma indicação prévia da utilização do espaço 
  de nomes standard std.

	using namespace std;

  Isto evita termos comandos tais como:

 	std::cout << "Viva" << std::endl;
*/


int main()
{
	cout << endl << "\tViva Mundo!" << endl;

	cout << endl << endl;

	system("pause");
	return 0;
}
