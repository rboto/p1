#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

// Programa Ref_042 pointers3_ptr_to_ptr_1

// Pointers to Pointers
//
// Ponteiros que apontam para ponteiros
//
// São vars-ponteiros que guardam endereços de outras vars-ponteiros
// que, por sua vez, guardam endereços de vars finais, permitindo assim 
// a estas um acesso de dupla indirecção. 
// 
// Particularidades:
//
// ** significa dupla indirecção/ponteiro para ponteiro
//

// Exemplo

int main()
{
	setlocale(LC_ALL, "Portuguese");
	
	int val;
	int *p1;  // ponteiro indirecção simples(single ptr)
	int **p2; // ponteiro indirecção dupla (double ptr)
	
	val = 5;

	p1 = &val;
	p2 = &p1;

	printf("\n End da var val = %ld", (unsigned long) &val);
	printf("\n End do ptr p1 = %ld", (unsigned long) &p1);
	printf("\n End do ptr p2 = %ld", (unsigned long) &p2);
	printf("\n Valor do ptr p1 = %ld", (unsigned long) p1);
	printf("\n Valor do ptr p2 = %ld", (unsigned long) p2);

	printf("\n Acesso directo a var val: %d", val);
	printf("\n Acesso indirecto simples a var val: %d", *p1);
	printf("\n Acesso indirecto duplo a var val: %d", **p2);

	**p2 = 30;
	
	printf("\n Acesso directo a var val: %d" , val);
	printf("\n Acesso indirecto simples a var val: %d", *p1);
	printf("\n Acesso indirecto duplo a var val: %d", **p2);

	printf("\n\n");
	system("pause");

	return 0;
}
