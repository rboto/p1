#include <stdio.h>
#include <stdlib.h>

// Programa C: Ref_014_chars_1

// Análise e tratamento de carateres individuais

// Este programa pede para o utilizador digitar um carater
// qualquer e a seguir indica o tipo de carater, ou seja,
// dígito (ex.= 2, 6), carater normal (ex.= A, a, F) ou especial 
// (ex.:$,%,#,?)

// Relembrar que os carateres são armazenados na memória através
// de códigos padrão.
// Por exemplo, em decimal, no Código ASCII: a letra A=65, 
// B=66, Z=90 e a=97.
// Por sua vez, o número 0=48 e 1=49;
// O carater especial #=35, etc.
// Cada carater ocupa um byte de Memória (RAM)

int main()
{
	char ch;
	
	printf("\n\nDigite uma carater: ");
	fflush(stdin);
	scanf("%c", &ch);
	
	if ( (ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z') )
	{
		printf("\n\nTrata-se de um carater alfanumerico");
	}
	else if ( ch >= '0' && ch <= '9')
	{
		printf("\n\nTrata-se de um digito");
	}
	else
	{
		printf("\n\nTrata-se de um carater especial");
	}

	printf("\n\n");
	system("PAUSE");
	
	return 0;
}
