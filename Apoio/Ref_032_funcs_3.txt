#include <stdio.h>

// Este programa revê as strings e usa funcs como strings 
// como argumentos.

// Uma string é uma cadeia de carateres terminados pelo
// carater especial '\0' (inserido automaticamente no
// fim da string).

// A string é assim um array ou vector de carateres, no
// qual cada elemnto é acedido através do nome da string
// e de um índice de posição.
//
// As strings quando inicializadas devem estar delineadas
// por aspas, exemplo
//
// char msg[] = "o rio tejo"
//
// Atenção que os carateres individais de uma str devem 
// ser referenciados por plicas, exemplo, 'a'


// Função conta e devolve a quantidade de espaços 
// contidos numa string
int conta_espacos(char str1[])
{
	int i; // índice geral
	int k; // contador de espaços

	i = 0;
	k = 0;
	while(str1[i] != '\0')
	{
		if (str1[i] == ' ')
		{
			k++;
		}
	
		i++;
	}

	return k;
} // fecha conta_espacos


// Esta função conta e devolve a quantidade de letras 
// maiusculas numa string
int conta_maiusculas(char texto[])
{
	int i;
	int k; // ctdr de letras maiusculas

	i = 0;
	k = 0;
	while (texto[i] != '\0')
	{
		if (texto[i] >= 'A' && texto[i] <= 'Z')
		{
			k++;
		}
		i++;
	}

	return k;	
}



// Esta função inverte e imprime uma string
void inverte_e_imprime_str(char str[])
{
	int i; 

	printf("\n\n\n\nEis a nova string a inverter: %s\n\n" , str);

// percorrer a string até ao fim
	i = 0;
	while (str[i] != '\0')
	{
		i++;
	}

// aqui, na string estamos posicionados no carater '\0'

	i--;  // recua da '\0' para o último carater válido da string

// neste momento a var i está posicionada no fim da string (antes da '\0')
	printf("\nEis a string invertida: \n\n");
	while (i >= 0)
	{
		printf("%c", str[i]);
		i--; // recua para carater anterior
	}
}

// Experimentos com funções e strings
int main()
{
	char info[40];
	int num_espacos;

	printf("\n\nDigite uma string:\n\n");
	fflush(stdin);
	gets(info);

// chamar uma função para contar o número de espaços

// chamar uma função para contar o numero de letras maiusculas

// chamar uma func para inverter e imprimir uma string

	inverte_e_imprime_str("o rio tejo");

}
